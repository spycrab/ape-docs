139  [[maybe_unused]] u16 seg_val = PrefixToValue(prefix);
140 
141  if constexpr (std::is_same<T, i8>::value) {
142  return static_cast<i8>(ParameterTo<u8>(parameter, prefix));
143  } else if constexpr (std::is_same<T, i8&>::value) {
144  return *reinterpret_cast<i8*>(&ParameterTo<u8&>(parameter, prefix));
145  } else if constexpr (std::is_same<T, i16>::value) {
146  return static_cast<i16>(ParameterTo<u16>(parameter, prefix));
147  } else if constexpr (std::is_same<T, i16&>::value) {
148  return *reinterpret_cast<i16*>(&ParameterTo<u16&>(parameter, prefix));
149  } else {
150  static_assert(std::is_same<T, u8>::value || std::is_same<T, u8&>::value ||
151  std::is_same<T, u16>::value ||
152  std::is_same<T, u16&>::value,
153  "Bad type provided!");
154  }
155 
156  using PType = Instruction::Parameter::Type;
157  if constexpr (std::is_same<T, u8>::value || std::is_same<T, u8&>::value) {
158  if (parameter.IsWord()) {
159  LOG("[BYTE] Tried to convert a WORD parameter to BYTE for " +
160  parameter.ToString(prefix) + "!");
161  throw ParameterLengthMismatchException(parameter);
162  }
163 
164  switch (parameter.GetType()) {
165  case PType::AL:
166  return AL;
167  case PType::AH:
168  return AH;
169 
170  case PType::BL:
171  return BL;
172  case PType::BH:

182  case PType::DH:

184 
185  case PType::Value_WordAddress:
186  return Memory::Get<u8>(seg_val, parameter.GetData<u16>());
187  case PType::Value_BP_Offset:
188  return Memory::Get<u8>(seg_val, BP + parameter.GetData<u8>());
189  case PType::Value_BP_WordOffset:
190  return Memory::Get<u8>(seg_val, BP + parameter.GetData<u16>());
191  case PType::Value_BP_DI:
192  return Memory::Get<u8>(seg_val, BP + DI);
193  case PType::Value_BP_DI_Offset:
194  return Memory::Get<u8>(seg_val, BP + DI + parameter.GetData<u8>());
195  case PType::Value_BP_DI_WordOffset:
196  return Memory::Get<u8>(seg_val, BP + DI + parameter.GetData<u16>());
197  case PType::Value_BP_SI:
198  return Memory::Get<u8>(seg_val, BP + SI);
199  case PType::Value_BP_SI_Offset:
200  return Memory::Get<u8>(seg_val, BP + SI + parameter.GetData<u8>());
201  case PType::Value_BP_SI_WordOffset:
202  return Memory::Get<u8>(seg_val, BP + SI + parameter.GetData<u16>());
203  case PType::Value_BX:

221  case PType::Value_DI:
222  return Memory::Get<u8>(seg_val, DI);
223  case PType::Value_DI_Offset:
224  return Memory::Get<u8>(seg_val, DI + parameter.GetData<u8>());
225  case PType::Value_DI_WordOffset:
226  return Memory::Get<u8>(seg_val, DI + parameter.GetData<u16>());
227  case PType::Value_SI:
228  return Memory::Get<u8>(seg_val, SI);
229  case PType::Value_SI_Offset:
230  return Memory::Get<u8>(seg_val, SI + parameter.GetData<u8>());
231  case PType::Value_SI_WordOffset:
232  return Memory::Get<u8>(seg_val, SI + parameter.GetData<u16>());
233  default:
